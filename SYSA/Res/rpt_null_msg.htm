提示您：ReportCls对象还未初始化。


 说明：调用ReportCls类，需要实现onReportInit事件与app_oncreate事件。

 实例如下：



 
   <!--#include file="../sdk/reportcls.asp"-->
   <%
	'初始化报表基础配置
	'rpt：是reportcls对象, state：加载类型，0=初始化加载， 1=回调加载
	sub onReportInit(byval rpt, byval  state)
		'此处添加您的代码, 例如
		'rpt.title = "xxx报表"
		'rpt.sql = "xxx"
		'............
	end sub
	
	'初始化表格列、数据格式等配置
	'lvw：报表的listview子对象
	sub app_oncreate(byval lvw)
		'此处添加您的代码, 例如
		'lvw.currsum = true '开启当前页求和
		'lvw.currsum = true '开启所有页求和
		'lvw.headers(1).dbtype = "number" '设置第一列格式为数字
		'......
	end sub
   %>
